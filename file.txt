ğŸ“Œ PROCESS CSV DATA (Example)

Find average marks:

import csv

total = 0
count = 0

with open("data.csv", "r") as f:
    reader = csv.DictReader(f)
    for row in reader:
        total += int(row["marks"])
        count += 1

average = total / count

ğŸ“Œ WRITE CSV OUTPUT
import csv

with open("output.csv", "w", newline="") as f:
    writer = csv.writer(f)
    writer.writerow(["Average Marks"])
    writer.writerow([average])
